class interface
	DS_BILINKED_LIST [G -> ]

create 
	make,
	make_equal,
	make_from_linear,
	make_from_array,
	make_default

feature -- Access

	new_cursor: DS_BILINKED_LIST_CURSOR [G]
			-- New external cursor for traversal
	
feature -- Element change

	put_left_cursor (v: G; a_cursor: like new_cursor)
			-- Add v to left of a_cursor position.
			-- Do not move cursors.
			-- (Synonym of a_cursor.put_left (v).)
			-- (Performance: O(1).)
			-- Was declared in {DS_BILINKED_LIST} as synonym of `force_left_cursor`.

	force_left_cursor (v: G; a_cursor: like new_cursor)
			-- Add v to left of a_cursor position.
			-- Do not move cursors.
			-- (Synonym of a_cursor.put_left (v).)
			-- (Performance: O(1).)
			-- Was declared in {DS_BILINKED_LIST} as synonym of `put_left_cursor`.

	extend_left_cursor (other: DS_LINEAR [G]; a_cursor: like new_cursor)
			-- Add items of other to left of a_cursor position.
			-- Keep items of other in the same order.
			-- Do not move cursors.
			-- (Synonym of a_cursor.extend_left (other).)
			-- (Performance: O(other.count).)
			-- Was declared in {DS_BILINKED_LIST} as synonym of `append_left_cursor`.

	append_left_cursor (other: DS_LINEAR [G]; a_cursor: like new_cursor)
			-- Add items of other to left of a_cursor position.
			-- Keep items of other in the same order.
			-- Do not move cursors.
			-- (Synonym of a_cursor.extend_left (other).)
			-- (Performance: O(other.count).)
			-- Was declared in {DS_BILINKED_LIST} as synonym of `extend_left_cursor`.
	
feature -- Removal

	remove_last
			-- Remove item at end of list.
			-- Move any cursors at this position `forth`.
			-- (Performance: O(1).)

	remove_at_cursor (a_cursor: like new_cursor)
			-- Remove item at a_cursor position.
			-- Move any cursors at this position `forth`.
			-- (Synonym of a_cursor.remove.)
			-- (Performance: O(1).)

	remove_left_cursor (a_cursor: like new_cursor)
			-- Remove item to left of a_cursor position.
			-- Move any cursors at this position `forth`.
			-- (Synonym of a_cursor.remove_left.)
			-- (Performance: O(1).)

	prune_last (n: INTEGER_32)
			-- Remove n last items from list.
			-- Move all cursors `off`.
			-- (Performance: O(n).)

	prune_left_cursor (n: INTEGER_32; a_cursor: like new_cursor)
			-- Remove n items to left of a_cursor position.
			-- Move all cursors `off`.
			-- (Synonym of a_cursor.prune_left (n).)
			-- (Performance: O(n).)

	keep_last (n: INTEGER_32)
			-- Keep n last items of list.
			-- Move all cursors `off`.
			-- (Performance: O(n).)
	
invariant
	first_constraint: attached first_cell as l_first_cell implies l_first_cell.left = Void

end -- class DS_BILINKED_LIST
