deferred class interface
	DS_EXTENDIBLE [G -> ]

feature -- Status report

	extendible (n: INTEGER_32): BOOLEAN
			-- May container be extended with n items?
		require
			positive_n: n >= 0
	
feature -- Element change

	put (v: G)
			-- Add v to container.
		require
			extendible: extendible (1)
		ensure
			added: has (v)

	force (v: G)
			-- Add v to container.
		ensure
			added: has (v)

	extend (other: DS_LINEAR [G])
			-- Add items of other to container.
			-- Add other.first first, etc.
		require
			other_not_void: other /= Void
			extendible: extendible (other.count)

	append (other: DS_LINEAR [G])
			-- Add items of other to container.
			-- Add other.first first, etc.
		require
			other_not_void: other /= Void
	
end -- class DS_EXTENDIBLE
