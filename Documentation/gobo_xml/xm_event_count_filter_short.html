class interface
	XM_EVENT_COUNT_FILTER

create 
	make_null,
	make_next

feature -- Count

	processing_instructions: INTEGER_32
			-- Number of PI events.

	comments: INTEGER_32
			-- Number of comment events.

	start_tags: INTEGER_32
			-- Number of start tag events.

	attributes: INTEGER_32
			-- Number of attribute events.

	end_tags: INTEGER_32
			-- Number of end tag events.
			-- (equal to start tags on well formed documents)

	contents: INTEGER_32
			-- Number of content events.

	events: INTEGER_32
			-- Total number of events.
		ensure
			definition: Result = (comments + processing_instructions + start_tags + attributes + end_tags + contents)
	
feature -- Events

	on_start
			-- Reset counters.

	on_processing_instruction (a_name: STRING_8; a_content: STRING_8)
			-- Count and forward.

	on_comment (a_content: STRING_8)
			-- Count and forward.

	on_start_tag (a_namespace: detachable STRING_8; a_prefix: detachable STRING_8; a_local_part: STRING_8)
			-- Count and forward.

	on_attribute (a_namespace: detachable STRING_8; a_prefix: detachable STRING_8; a_local_part: STRING_8; a_value: STRING_8)
			-- Count and forward.

	on_end_tag (a_namespace: detachable STRING_8; a_prefix: detachable STRING_8; a_local_part: STRING_8)
			-- Count and forward.

	on_content (a_content: STRING_8)
			-- Count and forward.
	
end -- class XM_EVENT_COUNT_FILTER
