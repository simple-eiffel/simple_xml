	callbacks_pipe (a: ARRAY [XM_CALLBACKS_FILTER]): XM_CALLBACKS
			-- Make a pipe,

			-- return first item of pipe.
		require
			a_not_void: a /= Void
			a_not_empty: a.count > 0
		ensure
			pipe_not_void: Result /= Void

		require
			a_not_void: a /= Void
		ensure
			pipe_not_void: Result /= Void
