note
	description: "Entity definition and scanner"
	remark: "This class represents an entity name/type and also a scanner for this entity. The two concepts could be distinct, but this helps enforcing the property that an entity cannot be recursively included, as the scanner can be checked for being in use if the entity is recursively included. This is not really a is-a relationship though."
	library: "Gobo Eiffel XML Library"
	copyright: "Copyright (c) 2002-2014, Eric Bezault and others"
	license: "MIT License"
	date: "$Date$"
	revision: "$Revision$"

class interface
	XM_EIFFEL_ENTITY_DEF

create 
	make_literal,
	make_external,
	make_def

feature -- Resolver

	resolver: XM_EXTERNAL_RESOLVER
	
feature -- Status report

	is_external: BOOLEAN
			-- Is current entity an external entity?

	is_literal: BOOLEAN
			-- Is current entity a literal entity?
		ensure
			definition: Result = not is_external
	
feature -- Access

	literal_name: detachable STRING_8
			-- Literal entity name

	value: detachable STRING_8
			-- Literal entity value

	external_id: detachable XM_DTD_EXTERNAL_ID
			-- Resolve external value.

	hash_code: INTEGER_32
			-- Hash code value
			-- (on the entity name aspect)
	
feature -- Scanner: set input buffer

	apply_input_buffer
			-- Set input buffer.
	
feature -- Scanner: events

	in_use: BOOLEAN
			-- Has the first token been processed?

	read_token
			-- Ignore XML declaration that may be at the start of an external entity.
			-- Also that places the content of an external DTD within
			-- a (DOCTYPE_DECL_START, DOCTYPE_DECL_END) token pair.
	
feature -- Scanner: initialization

	reset
			-- Reset first seen status.
	
invariant
	value_not_void: value /= Void or external_id /= Void
	value_name_void_consistent: (value = Void) = (literal_name = Void)
	resolver_not_void: resolver /= Void
	type: is_literal xor is_external

end -- class XM_EIFFEL_ENTITY_DEF
