class interface
	XM_NAMESPACE

create 
	make,
	make_default

feature -- Access

	ns_prefix: detachable STRING_8
			-- Prefix of current namespace

	uri: STRING_8
			-- Namespace URI
	
feature -- Status report

	is_equal (other: like Current): BOOLEAN
			-- Are the two namespaces equal?
		ensure then
			definition: Result = String_.same_string (uri, other.uri)

	hash_code: INTEGER_32
			-- Hash code of URI

	out: STRING_8
			-- Out
	
feature -- Status report

	same_prefix (other: XM_NAMESPACE): BOOLEAN
			-- Same
		require
			other_not_void: other /= Void
		ensure
			equal: Result implies is_equal (other)
			same_prefix: Result implies (ns_prefix = other.ns_prefix or else ((attached ns_prefix as l_ns_prefix and attached other.ns_prefix as l_other_ns_prefix) and then String_.same_string (l_ns_prefix, l_other_ns_prefix)))

	has_prefix: BOOLEAN
			-- Is there an explicit prefix?
			-- (not a default namespace declaration)
		ensure
			definition: Result = (attached ns_prefix as l_ns_prefix and then l_ns_prefix.count > 0)
	
invariant
	uri_not_void: uri /= Void

end -- class XM_NAMESPACE
