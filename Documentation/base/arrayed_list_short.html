feature -- Initialization

	make (n: INTEGER_32)
			-- Allocate list with n items.
			-- (n may be zero for empty list.)
		require
			valid_number_of_items: n >= 0
		ensure
			correct_position: before
			is_empty: is_empty

	make_filled (n: INTEGER_32)
			-- Allocate list with n items.
			-- (n may be zero for empty list.)
			-- This list will be full.
		require
			valid_number_of_items: n >= 0
			has_default: ({G}).has_default
		ensure
			correct_position: before
			filled: full
